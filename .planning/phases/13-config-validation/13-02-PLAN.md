**src/soma/cross_session.py** — Per D-14, extends PressurePredictor:

```python
from __future__ import annotations
from soma.predictor import PressurePredictor, Prediction
from soma.session_store import load_sessions, SessionRecord
import math

class CrossSessionPredictor(PressurePredictor):
    def __init__(self, window: int = 10, horizon: int = 5) -> None:
        super().__init__(window=window, horizon=horizon)
        self._session_patterns: list[list[float]] = []

    def load_history(self, base_dir=None) -> None:
        records = load_sessions(base_dir=base_dir)
        for r in records:
            if len(r.pressure_trajectory) >= 5:  # min trajectory length
                self._session_patterns.append(r.pressure_trajectory)

    def predict(self, next_threshold: float) -> Prediction:
        base = super().predict(next_threshold)
        if len(self._session_patterns) < 3 or len(self._pressures) < 3:
            return base
        # Find most similar past trajectory prefix
        current = list(self._pressures)[-min(len(self._pressures), self.window):]
        best_score, best_cont = 0.0, None
        for past in self._session_patterns:
            for start in range(len(past) - len(current) - self.horizon):
                segment = past[start:start + len(current)]
                sim = _cosine_similarity(current, segment)
                if sim > best_score and sim > 0.8:
                    best_score = sim
                    best_cont = past[start + len(current):start + len(current) + self.horizon]
        if best_cont:
            cross_pred = max(best_cont)
            blended = 0.6 * base.predicted_pressure + 0.4 * cross_pred
            confidence = base.confidence * 0.6 + best_score * 0.4
            return Prediction(
                current_pressure=base.current_pressure,
                predicted_pressure=min(1.0, max(0.0, blended)),
                actions_ahead=base.actions_ahead,
                will_escalate=blended >= next_threshold and confidence > 0.3,
                next_threshold=next_threshold,
                dominant_reason="cross_session" if cross_pred > base.predicted_pressure else base.dominant_reason,
                confidence=confidence,
            )
        return base

    def to_dict(self) -> dict:
        d = super().to_dict()
        d["session_patterns"] = self._session_patterns
        return d

    @classmethod
    def from_dict(cls, data: dict) -> CrossSessionPredictor:
        obj = cls(window=data.get("window", 10), horizon=data.get("horizon", 5))
        # restore base state
        for p in data.get("pressures", []):
            obj._pressures.append(p)
        for a in data.get("action_log", []):
            obj._action_log.append(a)
        obj._session_patterns = data.get("session_patterns", [])
        return obj
```

**src/soma/threshold_tuner.py** — Per D-13, percentile-based statistical optimization. No sklearn, no ML:

```python
from __future__ import annotations
import statistics
from soma.types import ResponseMode

def compute_optimal_thresholds(
    run_results: list,  # list of BenchmarkMetrics-like objects with per_action data
    target_false_positive_rate: float = 0.05,
) -> dict[str, float]:
    """Find guide/warn/block thresholds from benchmark data.

    Collects pressure at GUIDE+ events. Separates true positives (error in next 3 actions)
    from false positives. Sets guide threshold at percentile that achieves target FP rate.
    warn = guide + 0.25, block = guide + 0.50.

    Returns DEFAULT_THRESHOLDS if insufficient data.
    """
```

Implementation:
- Iterate `run_results`. For each, iterate `per_action` list (dicts with pressure, mode, etc.)
- For each action where mode is GUIDE/WARN/BLOCK: check if any of next 3 actions have `error=True`
  - Yes → true_positive_pressures.append(pressure)
  - No → false_positive_pressures.append(pressure)
- If false_positive_pressures is empty: return `{"guide": 0.25, "warn": 0.50, "block": 0.75}`
- Sort false_positive_pressures. Find index at `(1.0 - target_false_positive_rate)` percentile
- `guide = false_positive_pressures[idx]`
- Clamp guide between 0.10 and 0.60 (safety bounds)
- `warn = min(guide + 0.25, 0.90)`, `block = min(guide + 0.50, 0.95)`
- Return `{"guide": guide, "warn": warn, "block": block}`

```python
from __future__ import annotations
from soma.vitals import compute_drift
from soma.types import Action

PHASE_WEIGHTS: dict[str, dict[str, float]] = {
    "research": {"Read": 0.4, "Grep": 0.3, "Glob": 0.2, "Bash": 0.1},
    "implement": {"Edit": 0.4, "Write": 0.3, "Read": 0.2, "Bash": 0.1},
    "test": {"Bash": 0.5, "Read": 0.3, "Edit": 0.2},
    "debug": {"Bash": 0.3, "Read": 0.3, "Edit": 0.2, "Grep": 0.2},
}

def compute_phase_aware_drift(
    actions: list[Action],
    baseline_vector: list[float],
    known_tools: list[str],
    current_phase: str,
) -> float:
    """Drift that accounts for expected tool distribution in current task phase.

    If tools match expected phase pattern, drift is reduced by up to 50%.
    If phase is unknown or tools don't match, raw drift returned unchanged.
    """
```

Implementation:
- Compute `raw_drift = compute_drift(actions, baseline_vector, known_tools)`
- Look up `phase_pattern = PHASE_WEIGHTS.get(current_phase, {})`
- If no pattern: return raw_drift
- Compute actual tool distribution from last N actions
- Compute phase_alignment: for each tool in phase_pattern, `min(actual_pct / expected_pct, 1.0)`, average across tools
- Return `raw_drift * (1.0 - 0.5 * phase_alignment)` — max 50% reduction
